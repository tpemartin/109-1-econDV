```{r subsetDataTWbank}
subsetDataTWbank = {
  big5bankInterestRates %>%
    mutate(
       年月日={
         191100+as.integer(年月) -> mockDate
         paste0(stringr::str_sub(mockDate,1,4),"-",str_sub(mockDate,5,6),"-01") # 加了01日
       },
       西元年月=lubridate::ymd(年月日)
    )  %>%
    select(
      "銀行", "西元年月", matches("定存利率-([一三六]個月|[一二三]年期)-固定")
    ) %>%
    filter(
      銀行=="臺灣銀行"
    )
}
```
